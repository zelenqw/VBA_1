def rmse_on_alpha(reg_class, alpha_values):
    train_mse = []
    test_mse = []
    for alpha in alpha_values:
        reg = reg_class(alpha=alpha)
        reg.fit(X_train, y_train)
        
        train_mse.append((mean_squared_error(y_train, reg.predict(X_train)))**(1/2))
        test_mse.append((mean_squared_error(y_test, reg.predict(X_test)))**(1/2))
        
    plt.figure()
    plt.plot(alpha_values, test_mse)
    plt.xlabel('alpha')
    plt.ylabel('error')
